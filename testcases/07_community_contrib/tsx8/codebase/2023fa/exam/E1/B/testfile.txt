 *   Read 4 integers; sum non-negative into x, negatives into y; print x then y.
 */

int main() {
    int x;
    int y;
    int i;
    int t;
    x = 0;
    y = 0;
    for (i = 0; i < 4; i = i + 1) {
        t = getint();
        if (t < 0) {
            y = y + t;
        } else {
            x = x + t;
        }
    }
    printf("%d\n%d", x, y);
    return 0;
}
