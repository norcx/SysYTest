/* Description:
 * Print all subsets of {0..n-1} using incremental construction.
 */

int A[10];

void print_subset(int n, int A[], int cur) {
  int i;
  for (i = 0; i < cur; i = i + 1) {
    printf("%d ", A[i]);
  }
  printf("\n");

  int s;
  if (cur) {
    s = A[cur - 1] + 1;
  } else {
    s = 0;
  }

  for (i = s; i < n; i = i + 1) {
    A[cur] = i;
    print_subset(n, A, cur + 1);
  }
}

int main() {
  int n;
  n = getint();
  print_subset(n, A, 0);
  return 0;
}
