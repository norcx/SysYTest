/* Description:
 *   Converted from a Windows stress-test harness that prints iteration counts.
 *   SysY cannot invoke external executables or compare files, so the original
 *   "break when outputs differ" condition is simulated by stopping at a
 *   user-provided iteration k (k == 0 means never stop early).
 *
 *   Input (one integer per line):
 *     n : max iterations (treated as |n|)
 *     k : stop iteration (treated as |k|), 0 or 1..n
 *
 *   Output:
 *     Prints 1..min(n, k) each on its own line (or 1..n if k == 0).
 */

int main() {
    int n;
    int k;
    int i;
    int cnt;

    n = getint();
    k = getint();
    if (n < 0) n = 0 - n;
    if (k < 0) k = 0 - k;

    cnt = 0;
    for (i = 0; i < n; i = i + 1) {
        cnt = cnt + 1;
        printf("%d\n", cnt);
        if (k != 0 && cnt == k) {
            break;
        }
    }
    return 0;
}
