/* Description:
 *   Differential-test style harness: compares two implementations on a stream
 *   of testcases; prints the number of consecutive matches and stops at the
 *   first mismatch (mirrors the original Windows shell-based fuzzer loop).
 */

int cnt;

int calc1(int a, int b, int c) {
    return a * b + c;
}

int calc2(int a, int b, int c) {
    int res;
    res = a * b;
    if (c != 0) {
        res = res - c;
    } else {
        res = res + c;
    }
    return res;
}

int main() {
    int a;
    int b;
    int c;
    int r1;
    int r2;

    for (;;) {
        a = getint();
        if (a == 0) {
            break;
        }
        b = getint();
        c = getint();

        r1 = calc1(a, b, c);
        r2 = calc2(a, b, c);
        if (r1 != r2) {
            break;
        }

        cnt = cnt + 1;
        printf("%d\n", cnt);
    }

    return 0;
}
