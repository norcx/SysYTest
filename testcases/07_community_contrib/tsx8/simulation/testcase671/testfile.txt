/* Description:
 *   Reads n and m, then reads m triples (x, y, z).
 *   Outputs a single line containing m '0' characters, then '\n'.
 */

int n;
int m;

int main() {
    int i;
    int x;
    int y;
    int z;

    n = getint();
    m = getint();

    for (i = 1; i <= m; i = i + 1) {
        x = getint();
        y = getint();
        z = getint();
    }

    for (i = 1; i <= m; i = i + 1) {
        printf("0");
    }
    printf("\n");
    return 0;
}
