/* Description:
 *   For each test case, read n and m and print:
 *     n - m - 2 if n != 0 and m != 0, else n - m.
 */

int main() {
  int T;
  int t;
  int n;
  int m;

  T = getint();
  for (t = 0; t < T; t = t + 1) {
    n = getint();
    m = getint();
    if (n != 0 && m != 0) {
      printf("%d\n", n - m - 2);
    } else {
      printf("%d\n", n - m);
    }
  }
  return 0;
}
