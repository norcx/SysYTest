const int MAXN = 1000;

int val[MAXN + 5];

int main() {
  int n;
  int m;
  int i;
  int x;
  int y;
  int t;
  int ans;

  n = getint();
  m = getint();

  for (i = 1; i <= n; i = i + 1) {
    val[i] = getint();
  }

  ans = 0;
  for (i = 1; i <= m; i = i + 1) {
    x = getint();
    y = getint();
    t = val[x];
    if (val[y] < t) {
      t = val[y];
    }
    ans = ans + t;
  }

  printf("%d\n", ans);
  return 0;
}
