/* Description:
 * Convert the given C++ "stress test harness" into a deterministic SysY program.
 * The original code increments and prints a counter each loop iteration until a
 * mismatch is found by external tools; in SysY we model this as: read n, print
 * 1..n (one per line).
 */

int main() {
    int n;
    int cnt;
    int i;

    n = getint();
    cnt = 0;

    for (i = 0; i < n; i = i + 1) {
        cnt = cnt + 1;
        printf("%d\n", cnt);
    }

    return 0;
}
