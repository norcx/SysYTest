const int MOD = 998244353;
const int MAXP = 10000;

int n;
int p;
int dp[MAXP + 1];

int main() {
  int i;
  int j;
  n = getint();
  p = getint();

  dp[0] = 1;
  i = 1;
  for (i = 1; i <= n; i = i + 1) {
    if (i <= p) {
      for (j = p; j >= i; j = j - 1) {
        dp[j] = dp[j] + dp[j - i];
        if (dp[j] >= MOD) dp[j] = dp[j] - MOD;
      }
    }
  }

  printf("%d\n", dp[p]);
  return 0;
}
