/* Description:
 *   Adapted from a Windows/C++ fuzzing harness that repeatedly prints an
 *   iteration counter and stops when an external file-compare reports
 *   mismatch. Since Strict SysY cannot execute external programs, this
 *   testcase models the "mismatch flag" as an integer read from input:
 *   if flag != 0, stop; otherwise continue.
 */

int main() {
    int cnt;
    int flag;
    int i;

    cnt = 0;
    flag = 0;

    for (i = 0; i < 100; i = i + 1) {
        cnt = cnt + 1;
        printf("%d\n", cnt);
        flag = getint();
        if (flag != 0) {
            break;
        }
    }
    return 0;
}
