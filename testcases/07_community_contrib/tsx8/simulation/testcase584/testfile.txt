/* Description:
 *   This is a simplified, deterministic SysY version of a Windows
 *   differential-testing harness.
 *
 *   The original program prints an increasing counter each round and stops
 *   when an external file-compare command reports a difference.
 *
 *   In this SysY testcase, we model the "compare result" as input:
 *     - Read one integer per round.
 *     - If it is 0, continue.
 *     - If it is nonzero, stop.
 */

int main() {
    int cnt;
    int i;
    int diff;

    cnt = 0;
    diff = 0;
    for (i = 0; i < 100; i = i + 1) {
        cnt = cnt + 1;
        printf("%d\n", cnt);
        diff = getint();
        if (diff != 0) break;
    }
    return 0;
}
