/* Description:
 *   Converted from a Windows differential-testing harness that printed an
 *   incrementing counter until an external mismatch was detected.
 *
 *   SysY cannot invoke external programs, so this testcase parameterizes the
 *   stop point via input.
 *
 *   Input:
 *     n (>= 0): number of iterations before "mismatch"
 *   Output:
 *     1..n, each on its own line
 */

int main() {
    int n;
    int i;
    n = getint();
    for (i = 1; i <= n; i = i + 1) {
        printf("%d\n", i);
    }
    return 0;
}
