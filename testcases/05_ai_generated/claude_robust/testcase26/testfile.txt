// test26_continue_break_stress.sy - continue和break压力测试
int main() {
    int i, j, k;
    int count;
    
    // 测试1: 连续continue
    count = 0;
    for (i = 0; i < 10; i = i + 1) {
        if (i == 2) {
            continue;
        }
        if (i == 5) {
            continue;
        }
        if (i == 7) {
            continue;
        }
        count = count + 1;
    }
    printf("Test1 count=%d\n", count);  // 10-3=7
    
    // 测试2: continue后紧跟break条件
    count = 0;
    for (i = 0; i < 20; i = i + 1) {
        if (i % 2 == 0) {
            continue;
        }
        count = count + 1;
        if (count >= 5) {
            break;
        }
    }
    printf("Test2 count=%d i=%d\n", count, i);  // count=5, i=9
    
    // 测试3: 嵌套循环，内层break不影响外层
    count = 0;
    for (i = 0; i < 5; i = i + 1) {
        for (j = 0; j < 10; j = j + 1) {
            if (j == 3) {
                break;
            }
            count = count + 1;
        }
    }
    printf("Test3 count=%d\n", count);  // 5*3=15
    
    // 测试4: 嵌套循环，内层continue
    count = 0;
    for (i = 0; i < 5; i = i + 1) {
        for (j = 0; j < 5; j = j + 1) {
            if (j == 2) {
                continue;
            }
            count = count + 1;
        }
    }
    printf("Test4 count=%d\n", count);  // 5*4=20
    
    // 测试5: 三层嵌套，各层都有break/continue
    count = 0;
    for (i = 0; i < 4; i = i + 1) {
        if (i == 1) {
            continue;
        }
        for (j = 0; j < 4; j = j + 1) {
            if (j == 2) {
                break;
            }
            for (k = 0; k < 4; k = k + 1) {
                if (k == 1) {
                    continue;
                }
                count = count + 1;
            }
        }
    }
    printf("Test5 count=%d\n", count);  // 3个i * 2个j * 3个k = 18
    
    // 测试6: break在if-else中
    count = 0;
    for (i = 0; i < 100; i = i + 1) {
        if (i % 3 == 0) {
            count = count + 1;
        } else {
            if (count > 10) {
                break;
            }
        }
    }
    printf("Test6 count=%d i=%d\n", count, i);  // count=11, i=31
    
    return 0;
}
