// test24_static_array.sy - static数组测试
int get_next() {
    static int arr[5] = {10, 20, 30, 40, 50};
    static int idx = 0;
    int val;
    val = arr[idx];
    idx = idx + 1;
    if (idx >= 5) {
        idx = 0;
    }
    return val;
}

int accumulate() {
    static int history[10];
    static int count = 0;
    int val;
    val = getint();
    if (count < 10) {
        history[count] = val;
        count = count + 1;
    }
    int i;
    int sum = 0;
    for (i = 0; i < count; i = i + 1) {
        sum = sum + history[i];
    }
    return sum;
}

int main() {
    int i;
    
    printf("get_next sequence: ");
    for (i = 0; i < 12; i = i + 1) {
        printf("%d ", get_next());
    }
    printf("\n");
    // 预期: 10 20 30 40 50 10 20 30 40 50 10 20
    
    printf("Accumulate test:\n");
    for (i = 0; i < 5; i = i + 1) {
        printf("Sum so far: %d\n", accumulate());
    }
    
    return 0;
}
