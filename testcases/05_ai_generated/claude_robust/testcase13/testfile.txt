// test13_const_array.sy - const数组和const变量测试
const int SIZE = 5;
const int OFFSET = 10;
const int g_const_arr[5] = {1, 2, 3, 4, 5};

int sum_array(int arr[], int n) {
    int i;
    int sum = 0;
    for (i = 0; i < n; i = i + 1) {
        sum = sum + arr[i];
    }
    return sum;
}

int main() {
    const int LOCAL_SIZE = SIZE + 2;
    int arr[7] = {OFFSET, OFFSET + 1, OFFSET + 2, OFFSET + 3, OFFSET + 4, OFFSET + 5, OFFSET + 6};
    
    int i;
    printf("g_const_arr: ");
    for (i = 0; i < SIZE; i = i + 1) {
        printf("%d ", g_const_arr[i]);
    }
    printf("\n");
    
    printf("local arr: ");
    for (i = 0; i < LOCAL_SIZE; i = i + 1) {
        printf("%d ", arr[i]);
    }
    printf("\n");
    
    printf("sum of local arr: %d\n", sum_array(arr, LOCAL_SIZE));  // 10+11+12+13+14+15+16 = 91
    
    // const用于数组索引
    printf("arr[SIZE-1]=%d\n", arr[SIZE - 1]);  // arr[4] = 14
    printf("arr[OFFSET-SIZE]=%d\n", arr[OFFSET - SIZE]);  // arr[5] = 15
    
    return 0;
}
