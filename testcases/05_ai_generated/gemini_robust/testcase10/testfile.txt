// test10_matrix.sy
int main() {
    int A[4] = {1, 2, 3, 4}; // 2x2
    int B[4] = {5, 6, 7, 8}; // 2x2
    int C[4];
    int i, j, k;
    
    for (i = 0; i < 2; i = i + 1) {
        for (j = 0; j < 2; j = j + 1) {
            int sum = 0;
            for (k = 0; k < 2; k = k + 1) {
                // A[i][k] * B[k][j] -> A[i*2+k] * B[k*2+j]
                sum = sum + A[i * 2 + k] * B[k * 2 + j];
            }
            C[i * 2 + j] = sum;
        }
    }
    
    for (i = 0; i < 4; i = i + 1) {
        printf("%d ", C[i]);
    }
    return 0;
}
