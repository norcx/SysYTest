// test3_array.sy
const int LEN = 2 + 3;
int g_arr[LEN] = {1, 1 + 1, 1 * 3, 8 / 2, 10 % 5 + 5};

int main() {
    int i;
    int l_arr[5] = {10, 20, 30, 40, 50};
    for (i = 0; i < LEN; i = i + 1) {
        printf("%d ", g_arr[i]);
    }
    printf("\n");
    for (i = 0; i < 5; i = i + 1) {
        printf("%d ", l_arr[i]);
    }
    return 0;
}
