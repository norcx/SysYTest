int matrix[13000];
int a[100];

int transpose(int n, int matrix[], int rowsize){
    int colsize = n / rowsize;
    int i = 0;
    int j = 0;
    for(; i < colsize;){
        j = 0;
        for(; j < rowsize;){
            if (i < j){
                j = j + 1;
                continue;
            }
            int curr = matrix[i * rowsize + j];
            matrix[j * colsize + i] = matrix[i * rowsize + j];
            matrix[i * rowsize + j] = curr;
            j = j + 1;
        }
        i = i + 1;
    }
    return 0 - 1;
}

int main(){
    int n = getint();
    int len = getint();
    int i = 0;
    for(; i < len;){
        a[i] = getint();
        i = i + 1;
    }
    i = 0;
    for(; i < n;){
        matrix[i] = i;
        i = i + 1;
    }
    i = 0;
    for(; i < len;){
        transpose(n, matrix, a[i]);
        i = i + 1;
    }

    int ans = 0;
    i = 0;
    for(; i < len;){
        ans = ans + i * i * matrix[i];
        i = i + 1;
    }
    if (ans < 0) ans = 0 - ans;
    printf("%d", ans);
    printf("\n");
    return 0;
}
