int palindrome(int n)
{
    int a[4];
    int j;
    int flag;
    j=0;
    for(; j<4;)
    {
        a[j]=n%10;
        n=n/10;
        j=j+1;
    }

    if(a[0]==a[3] && a[1]==a[2])
    {
        flag=1;
    }else{
        flag=0;
    }
    return flag;
}

int main()
{
    int test;
    test=1221;
    int flag;
    flag=palindrome(test);
    if(flag==1)
        printf("%d", test);
    else
    {
        flag = 0;
        printf("%d", flag);
    }

    printf("\n");

    return 0;
}
