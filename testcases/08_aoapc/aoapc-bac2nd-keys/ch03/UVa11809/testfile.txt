int main() {
    int v;
    int i;
    int bestIdx;
    int bestDiff;
    int diff;

    for (;;) {
        v = getint();
        if (v == -1) {
            break;
        }

        bestIdx = 0;
        bestDiff = v - tableV[0];
        if (bestDiff < 0) {
            bestDiff = -bestDiff;
        }

        for (i = 1; i < N; i = i + 1) {
            diff = v - tableV[i];
            if (diff < 0) {
                diff = -diff;
            }
            if (diff < bestDiff) {
                bestDiff = diff;
                bestIdx = i;
            }
        }

        printf("%d %d\n", tableM[bestIdx], tableE[bestIdx]);
    }

    return 0;
}
