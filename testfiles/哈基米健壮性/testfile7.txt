void bubble_sort(int a[], int n) {
    int i, j;
    for (i = 0; i < n - 1; i = i + 1) {
        for (j = 0; j < n - i - 1; j = j + 1) {
            if (a[j] > a[j+1]) {
                int tmp = a[j];
                a[j] = a[j+1];
                a[j+1] = tmp;
            }
        }
    }
}

int main() {
    int data[5] = {5, 4, 3, 2, 1};
    bubble_sort(data, 5);
    int i;
    for (i = 0; i < 5; i = i + 1) {
        printf("%d ", data[i]);
    }
    return 0;
}
