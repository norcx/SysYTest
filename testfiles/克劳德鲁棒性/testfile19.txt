// test19_for_variants.sy - for循环各种变体测试
int main() {
    int i, j;
    int sum;
    
    // 基本for循环
    sum = 0;
    for (i = 0; i < 5; i = i + 1) {
        sum = sum + i;
    }
    printf("Basic for: sum=%d\n", sum);  // 0+1+2+3+4 = 10
    
    // 步长为2
    sum = 0;
    for (i = 0; i < 10; i = i + 2) {
        sum = sum + i;
    }
    printf("Step 2: sum=%d\n", sum);  // 0+2+4+6+8 = 20
    
    // 递减循环
    sum = 0;
    for (i = 10; i > 0; i = i - 1) {
        sum = sum + i;
    }
    printf("Decrement: sum=%d\n", sum);  // 10+9+...+1 = 55
    
    // 递减步长为3
    sum = 0;
    for (i = 15; i > 0; i = i - 3) {
        sum = sum + i;
    }
    printf("Decrement step 3: sum=%d\n", sum);  // 15+12+9+6+3 = 45
    
    // 嵌套for，内层依赖外层
    sum = 0;
    for (i = 1; i <= 5; i = i + 1) {
        for (j = 1; j <= i; j = j + 1) {
            sum = sum + 1;
        }
    }
    printf("Nested dependent: sum=%d\n", sum);  // 1+2+3+4+5 = 15
    
    // 循环变量在循环体内修改
    sum = 0;
    for (i = 0; i < 10; i = i + 1) {
        sum = sum + i;
        if (i == 5) {
            i = i + 2;  // 跳过6,7
        }
    }
    printf("Modified in body: sum=%d\n", sum);  // 0+1+2+3+4+5+8+9 = 32
    
    // 空循环体
    for (i = 0; i < 100; i = i + 1) {
    }
    printf("Empty body: i=%d\n", i);  // 100
    
    // 单次执行
    sum = 0;
    for (i = 0; i < 1; i = i + 1) {
        sum = 999;
    }
    printf("Single iteration: sum=%d\n", sum);  // 999
    
    // 零次执行
    sum = 0;
    for (i = 10; i < 5; i = i + 1) {
        sum = 999;
    }
    printf("Zero iteration: sum=%d\n", sum);  // 0
    
    return 0;
}
