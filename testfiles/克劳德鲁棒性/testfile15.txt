// test15_prime_sieve.sy - 埃拉托斯特尼筛法（数组+复杂控制流）
int is_prime[101];

void sieve(int n) {
    int i, j;
    // 初始化
    for (i = 0; i <= n; i = i + 1) {
        is_prime[i] = 1;
    }
    is_prime[0] = 0;
    is_prime[1] = 0;
    
    // 筛选
    for (i = 2; i * i <= n; i = i + 1) {
        if (is_prime[i]) {
            for (j = i * i; j <= n; j = j + i) {
                is_prime[j] = 0;
            }
        }
    }
}

int count_primes(int n) {
    int i;
    int cnt = 0;
    for (i = 2; i <= n; i = i + 1) {
        if (is_prime[i]) {
            cnt = cnt + 1;
        }
    }
    return cnt;
}

int main() {
    int n = getint();
    sieve(n);
    
    printf("Primes up to %d: ", n);
    int i;
    for (i = 2; i <= n; i = i + 1) {
        if (is_prime[i]) {
            printf("%d ", i);
        }
    }
    printf("\n");
    
    printf("Count: %d\n", count_primes(n));
    
    return 0;
}
