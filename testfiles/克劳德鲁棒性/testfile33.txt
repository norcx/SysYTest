// test33_selection_sort.sy - 选择排序
int arr[15];

int find_min_index(int a[], int start, int end) {
    int i;
    int min_idx;
    min_idx = start;
    for (i = start + 1; i < end; i = i + 1) {
        if (a[i] < a[min_idx]) {
            min_idx = i;
        }
    }
    return min_idx;
}

void selection_sort(int a[], int n) {
    int i;
    int min_idx;
    int tmp;
    for (i = 0; i < n - 1; i = i + 1) {
        min_idx = find_min_index(a, i, n);
        if (min_idx != i) {
            tmp = a[i];
            a[i] = a[min_idx];
            a[min_idx] = tmp;
        }
    }
}

void print_array(int a[], int n) {
    int i;
    for (i = 0; i < n; i = i + 1) {
        printf("%d ", a[i]);
    }
    printf("\n");
}

int main() {
    int n;
    int i;
    
    n = getint();
    for (i = 0; i < n; i = i + 1) {
        arr[i] = getint();
    }
    
    printf("Before: ");
    print_array(arr, n);
    
    selection_sort(arr, n);
    
    printf("After: ");
    print_array(arr, n);
    
    return 0;
}
